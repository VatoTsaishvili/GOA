function modifyDivContent() { const container = document.getElementById('container'); // ღილაკის წაშლა const button = document.getElementById('myButton'); if (button) { container.removeChild(button); } // პარაგრაფის ჩანაცვლება i თეგით const paragraph = document.getElementById('myParagraph'); if (paragraph) { const iTag = document.createElement('i'); const textNode = document.createTextNode('ეს არის i თეგი.'); iTag.appendChild(textNode); container.replaceChild(iTag, paragraph); } } // ფუნქციის გამოძახება გვერდის ჩატვირთვისას window.onload = modifyDivContent;
